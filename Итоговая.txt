try:
    a = float(input("Введите первое число:"))
    operation = input("Введите операцию (+, -, *, /):")
    b = float(input("Введите второе число:"))
    if operation == "+":
        result = a + b
    elif operation == "-":
        result = a - b
    elif operation == "*":
        result = a * b
    elif operation == "/":
        result = a / b
    else:
        print("Ошибка. Такой операции нет")
    print(f"Результат:{result}")
except ValueError:
    print(f"Ошибка: Невозможно преобразовать в число.")
except ZeroDivisionError:
    print("Ошибка: Деление на ноль невозможно!")

attempts = 0
start_time = datetime.now()
print(f"Игра началась в: {start_time.strftime('%H:%M:%S')}")

print("Начинаем! У вас 60 секунд.")
while True:
    try:
        guess = int(input(f"Попытка №{attempts + 1}. Ваше число:"))
        attempts += 1
        
        if guess < secret_number:
            print("Маловато...")
        elif guess > secret_number:
            print("Перебор!")
        elif guess = secret_number:
            print("🎉 Вы угадали!")
            break
        elapsed = (datetime.now() - start_time).seconds
        if elapsed >= 60:
            print(f"⏳ Время вышло! Загаданное число было: {secret_number}")
            break
            
            
    except Value_Error:
    print("Ошибка: введите целое число.")
    
end_time = datetime.now()
elapsed_seconds = (end_time - start_time).seconds

print(f"\n⏰ Конец игры: {end_time.strftime('%H:%M:%S')}")
print(f"⏱️  Вы потратили: {elapsed_seconds} секунд")
